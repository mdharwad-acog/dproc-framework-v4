Real-World Example: Company Analysis

Multi-step pipeline combining multiple APIs

export default async function analyze(inputs, context) { // 1. Fetch company from Crunchbase const company = await context.http.get( `https://api.crunchbase.com/companies/${inputs.name}` ); // 2. Get financials from SEC Edgar const financials = await context.http.get( `https://sec.gov/api/company/${company.ticker}` ); // 3. Analyze with LLM const analysis = await context.llm.generate({ prompt: `Analyze company: ${JSON.stringify(company)} Financials: ${JSON.stringify(financials)}` }); return { company, financials, analysis }; }

export default async function extractData(inputs, context) { // 1. Fetch raw document const doc = await context.http.get(inputs.url); // 2. Parse with LLM const extracted = await context.llm.generate({ model: 'claude-3-sonnet', prompt: `Extract from document:\n${doc}` }); // 3. Validate with Zod const validated = z.object({ entities: z.array(z.string()), relations: z.array(z.object({ from: z.string(), to: z.string() })) }).parse(JSON.parse(extracted)); return validated; }
